CREATE EXTENSION "uuid-ossp"


CREATE TABLE products (
id UUID DEFAULT uuid_generate_v4() PRIMARY KEY,
title VARCHAR(255) NOT NULL,
price INT NOT NULL,
description VARCHAR(255),
image VARCHAR(255) NOT NULL
);
